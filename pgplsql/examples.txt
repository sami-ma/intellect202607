CREATE FUNCTION func_sum(int[]) RETURNS int  
language plpgsql AS $$
DECLARE
s int := 0;
x int;
BEGIN
	FOREACH x IN ARRAY $1 LOOP
		s := s + x;
	END LOOP;
	RETURN s;
END;  $$ 

select func_sum( array [12, 6, 32, 7, 1])

CREATE OR REPLACE PROCEDURE exp(epno int)
    LANGUAGE 'plpgsql'
AS $BODY$
DECLARE
  erec emp%rowtype;
BEGIN
  select * into strict erec from emp where empno = epno;
  raise notice 'Employee Name: % Salary : %', erec.ename, erec.sal;
EXCEPTION
   when no_data_found then raise exception 'Invalid emlployee number : %', epno;
   when too_many_rows then raise exception 'Many Employees matches Employee number : %', epno;
END;
$BODY$;

call exp(1234)

$$
declare
  r record;

begin
  select * into strict r from emp where empno = $1;
  raise notice 'row is %', r;
exception
  when no_data_found then raise exception 'Invalid empno %', $1;
  when others then raise exception 'Something went wrong';
end;
 $$ ;

  x int;

 x := $1 / $2;

 $$ ;

CREATE OR REPLACE PROCEDURE cur()
    LANGUAGE 'plpgsql'
AS $BODY$
DECLARE
  erec  record;
  c1 cursor for select * from emp;
  S  emp.sal%type;
BEGIN
open c1;
loop
  fetch from c1 into erec;
  exit when not found;
  CASE  erec.deptno
	WHEN 10 THEN
		S = erec.sal + (erec.sal * 0.15);
	WHEN 20 THEN
		S = erec.sal + (erec.sal * 0.10);
	WHEN 30 THEN
		S = erec.sal + (erec.sal * 0.12);
	ELSE
		raise notice 'Department % not eligible for salary revision', erec.deptno;
  END CASE;
 raise notice  'EmpName : %  Deptno : %  Revised Salary : %', erec.ename, erec.deptno, S;
end loop;
close c1;
END;
$BODY$;

call cur()

create or replace function emp_trig1() returns trigger LANGUAGE plpgsql as
$$
begin
new.ename := upper(new.ename);
new.job := upper(new.job);
return new;
end; $$ ;

create trigger emptrig1
before update of ename, job  on emp for each row
execute procedure emp_trig1();

create trigger emptrig2
before insert  on emp for each row
execute procedure emp_trig1();
